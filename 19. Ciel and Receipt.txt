#include<iostream>
#include<math.h>

using namespace std;

int main() {
    int t, temp;
    cin>>t;  
    int menu[12];
    for (int i=1; i<=12; i++){
    	menu[i]=pow(2,i-1);
    }
    while(t--){
    	int a;
    	cin>>a;
    	temp=a/2048;
    	a=a%2048;
    	while(a!=0){
    		temp+=a%2;
    		a/=2;
    	}
    	cout<<temp<<endl;
    }

    return 0;
}
