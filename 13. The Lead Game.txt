#include<iostream>

using namespace std;

int main(){
    int t;
    int win=0, lead=0;
    int p_a=0, p_b=0;
    cin>>t;  
  	while(t--){
  		int a, b;
  		cin>>a>>b;
  		a+=p_a;
  		b+=p_b;
  		if(a>b){
  			if ((a-b)>lead){
  				lead=a-b;
  				win=1;
  			}
  		}
  		else {
  			if ((b-a)>lead){
  				lead=b-a;
  				win=2;
  			}
  		}
  		p_a=a;
  		p_b=b;
  	}
  	cout<<win<<" "<<lead;
   	
    return 0;
}
