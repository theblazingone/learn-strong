#include <iostream>
#include <array>

using namespace std;

int main(){
   int num;
   cin>>num;
   int n[num], m[num], add[num];
   for (int i=0; i<num; i++){
   	cin>>n[i]>>m[i];
   }
   for (int i=0; i<num; i++){
   	add[i] = n[i]+m[i];
   }
   for (int i=0; i<num; i++){
   	cout<<add[i]<<endl;
   }
   return 0;

}
