#include <iostream>

using namespace std;

int fact(int N);

int main() {
	int t;
	cin>>t;
	while(t--){
	    int n, result;
	    cin>>n;
	    result = fact(n);
	    cout<<result<<endl;
	}
	
	return 0;
}

int fact(int N){
    int f=1, i;
    for (int i=1; i<=N; i++){
        f=f*i;
    }
    
    return f;
}
