#include <iostream>

using namespace std;

int main(){
   int arr[200];
   int t;
   cin>>t;
   while (t--){
   	int n;
    cin>>n;
    arr[0]=1;
    int m=1, temp=0;
    for (int i=1; i<=n; i++){
    	for (int j=0; j<m; j++){
    	    int x = arr[j]*i+temp;
            arr[j]=x%10;
            temp=x/10;
    	}
    	while(temp>0){
    	    arr[m]=temp%10;
            temp/=10;
            m++;
    	}
    }
    for(int i=m-1; i>=0; i--){
    	cout<<arr[i];
    }
    cout<<endl;
   }

   return 0;
}
