#include <iostream>
using namespace std;

int main() {
    
	int t, cnt=0;
    int num;
    cin>>t;
    while(t--){
    	cin>>num;
    	int rem, rev=0;
    	while (num!=0){
    		rem = num%10;
    		rev = rev*10+rem;
    		num/=10;
    	}
    	cout<<rev<<endl;
    }

    return 0;
}
